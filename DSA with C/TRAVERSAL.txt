#include <stdio.h>
#include <stdlib.h>
typedef struct linked
{
    int data;
    struct linked *left;
    struct linked *right;
} link;
link *fun(int data)
{
    link *p = (link *)malloc(sizeof(link));
    p->data = data;
    p->left = NULL;
    p->right = NULL;
    return p;
}
// PRORDER TRAVERSAL
void preOrder(link *root)
{
    if (root != NULL)
    {
        printf("%d ", root->data); // ROOT-LEFT-RIGHT
        preOrder(root->left);
        preOrder(root->right);
    }
}
// POSTORDER TRAVERSAL
void postOrder(link *root)
{
    if (root != NULL)
    {
        postOrder(root->left); // LEFT-RIGHT-ROOT
        postOrder(root->right);
        printf("%d ", root->data);
    }
}
// INORDER TRAVERSAL
void inOrder(link *root)
{
    if (root != NULL)
    {
        inOrder(root->left); // LEFT-ROOT-RIGHT
        printf("%d ", root->data);
        inOrder(root->right);
    }
}
int isBST(link *p)
{
    static link *prev = NULL;
    if (p != NULL)
    {
        if (!isBST(p->left))
        {
            return 0; // FOR NOT BST
        }
        if (prev != NULL && p->data <= prev->data)
        {
            return 0;
        }
        prev = p;
        return isBST(p->right);
    }
    else
    {
        return 1; // FOR BST
    }
}
link *search(link *p, int key) // when we have to return any pointer valuse that time we make it pointer type
{
    if (p == NULL)
    {
        return NULL;
    }
    if (p->data == key)
    {
        return p;
    }
    else if (p->data > key)
    {
        return search(p->left, key);
    }
    else
    {
        return search(p->right, key);
    }
}
